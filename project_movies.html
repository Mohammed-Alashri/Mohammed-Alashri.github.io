cleaned_budget_avg <- clean_data_col%>%
   group_by(Genre)%>%
   summarise(avg_budget_m=mean(Budget))%>%
   arrange(-avg_budget_m) %>% 
   mutate( avg_genre_budget_m=avg_budget_m/1000000 ) %>% 
   subset(select=c(Genre, avg_genre_budget_m))
